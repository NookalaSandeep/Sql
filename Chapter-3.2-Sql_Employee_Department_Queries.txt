22.create new table with unique records?
========================================

create table unique_employee as
select * from (select *,row_number() over(partition by first_name,last_name order by first_name,last_name) rn from employee)employee where rn =1 order by emp_id

union     - drop duplicates from another table 
intersect - only matching records from both tables

union all
select emp_id,first_name,last_name,email,salary,experience_years,experience_months,dept_no from unique_employee
