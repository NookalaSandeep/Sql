1.Find the name,salary of each professor whose department is not equal to CSE and 
salary is greater than the smallest salary of the professor in CSE Department?
=================================================================================
select * from professor;
select * from department;

select p.Name,p.Salary from professor p
join department d 
on p.dept_no=d.dept_no
where d.dept_name != 'CSE'
and p.salary > (select min(salary) from professor where dept_no in (select dept_no from department where dept_name='CSE'))

2.Create table with uniq records from employee table using self join?
=====================================================================
select * from employee;

create table uniq_emp(
select e1.* from employee e1
join employee e2
on e1.fname =e2.fname
and e1.lname = e2.lname
and e1.id < e2.id );

============================================

4.ftech employees details who is earning max salary in his department?
=====================================================================
select * from uniq_emp;

select * from uniq_emp e1 
join
(select dept_no,max(salary) salary from uniq_emp group by dept_no) e2
on e1.dept_no = e2.dept_no
and e1.salary = e2.salary

5.Fetch the employees who are having same salary from other department?
=====================================================================
select * from uniq_emp e1 
join
uniq_emp e2
on e1.fname != e2.fname
and e1.lname != e2.lname
and e1.salary = e2.salary
and e1.dept_no != e2.dept_no

8. ftech student details with highest total marks?
==================================================
select * from student s
join
(select student_id,sum(marks) total from student group by student_id order by total desc limit 1) s1
on s.student_id = s1.student_id

======================================================

13.Total revenue by each product ?
=================================
select p.product_id,sum(p.unit_price * s.qty) revenue from products p
join sales s
on p.product_id = s.product_id
group by p.product_id
