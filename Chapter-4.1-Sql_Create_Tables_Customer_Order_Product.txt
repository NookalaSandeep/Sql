-- =========================
-- ORDERS TABLE (COMBINED)
-- =========================
CREATE TABLE orders (
    order_id INT,
    cust_id INT,
    prod_id INT,
    quantity INT,
    order_date DATE,
    PRIMARY KEY (order_id, prod_id),
    FOREIGN KEY (cust_id) REFERENCES customer(cust_id),
    FOREIGN KEY (prod_id) REFERENCES products(prod_id)
);
