create courses table:  (parent table):
--------------------------------------
CREATE TABLE COURSE (
COURSE_ID INT AUTO_INCREMENT,
COURSE_NAME VARCHAR(20) NOT NULL,
COURSE_FEE DEFAULT 0 CHECK(COURSE_FEE>= 0),    #CHECK CONTRAINT
COURSE_DURATION INT NOT NULL,                  #NOT NULL  CONTRAINT
PRIMARY KEY (COURSE_ID)                        #PRIMARY KEY CONTRAINT
);


create student table(child table):
---------------------------------
CREATE TABLE STUDENT(
STUDENT_ID INT AUTO_INCREMENT,
STUDENT_FNAME VARCHAR(50) NOT NULL ,
STUDENT_LNAME VARCHAR(50) NOT NULL,
STUDENT_AGE INT NOT NULL,
STUDENT_EMAIL VARCHAR(50) NOT NULL,
STUDENT_PHONE VARCHAR(20) NOT NULL,
STUDENT_ENROLLMENT_DATE TIMESTAMP NOT NULL,
STUDENT_COURSE_ID INT NOT NULL DEFAULT 1,                         #DEFAULT CONTRAINT
STUDENT_LOCATION VARCHAR(50) NOT NULL,
PRIMARY KEY (STUDENT_ID,STUDENT_FNAME),                           #COMPOSITE PRIMARY KEY CONTRAINT
UNIQUE KEY (STUDENT_EMAIL,STUDENT_PHONE),                         #UNIQUE KEY CONTRAINT
FOREIGN KEY (STUDENT_COURSE_ID) REFERENCES COURSE(COURSE_ID)      #FOREIGN KEY CONTRAINT
);

#Aggregate functions:
=====================

1.count:
--------
SELECT COUNT(*) AS TOTAL_STUDENT_COUNT FROM STUDENT;
SELECT COUNT(*) AS TOTAL_COURSES_COUNT FROM COURSE;

count & distinct:
-----------------
SELECT COUNT(STUDENT_COURSE_ID) AS TOTAL_COURSES FROM STUDENT;
SELECT COUNT(DISTINCT(STUDENT_COURSE_ID)) AS UNIQUE_COURSES FROM STUDENT;


group By:
---------
SELECT COUNT(*) AS NUM_STUDENTS_PER_COURSE,STUDENT_COURSE_ID FROM STUDENT GROUP BY STUDENT_COURSE_ID ;


Min & Max:
----------
SELECT MIN(STUDENT_AGE) FROM STUDENT;
SELECT MAX(STUDENT_AGE) FROM STUDENT;


sum:
----
SELECT SUM(COURSE_FEE) FROM COURSE;


Avg:
----
SELECT AVG(COURSE_FEE) FROM COURSE;


having:
-------
SELECT COUNT(*) AS NUM_STUDENTS_PER_COURSE,STUDENT_COURSE_ID FROM STUDENT GROUP BY STUDENT_COURSE_ID HAVING NUM_STUDENTS_PER_COURSE>1 ;


where:
------
SELECT * FROM STUDENT WHERE STUDENT_LNAME ='NOOKALA';


where and having:
-----------------

SELECT COUNT(*) AS NUM_STUDENTS_PER_COURSE,STUDENT_COURSE_ID FROM STUDENT WHERE STUDENT_LNAME ='NOOKALA' GROUP BY STUDENT_COURSE_ID HAVING NUM_STUDENTS_PER_COURSE>1 ;




logical operators:
==================

1. equal:  = 
------------
SELECT * FROM STUDENT WHERE STUDENT_LOCATION = 'NARSAMPET';


2. not equal: !=
---------------
SELECT * FROM STUDENT WHERE STUDENT_LOCATION != 'NARSAMPET';


3. and:
-------
SELECT * FROM STUDENT WHERE STUDENT_LOCATION = 'NARSAMPET' AND  STUDENT_LNAME = 'NOOKALA';


4.or:
-----
SELECT * FROM STUDENT WHERE STUDENT_LOCATION = 'NARSAMPET' OR  STUDENT_LNAME = 'NOOKALA';


between: (in range):
--------------------
select * from student where STUDENT_ID between 2 and 6


not between: (not range):
-------------------------
select * from student where STUDENT_ID not between 2 and 6;


in:(present within given list):
-------------------------------
select * from student where STUDENT_FNAME in ('sandeep','satish')


not in:(not present within given list):
---------------------------------------
select * from student where STUDENT_FNAME not in ('sandeep','satish')


DATE AND TIMESTAMP:
===================

DATETIME  format:  YYYY-MM-DD HH:MI: SS   Returns the current date.
TIMESTAMP format:  YYYY-MM-DD HH:MI: SS   Returns the current time.
YEAR 	  format:  YYYY or YY             Extracts the year part of a date  
DAY       format:  D                      Extracts the day part of a date  
MONTH     format:  M                      Extracts the month part of a date  
DATE	  format:  YYYY-MM-DD             Extracts the date part of a date 


DATE :

select now
select curdate()

------------------------------------------------
select extract(day from STUDENT_ENROLLMENT_DATE) from student;

DATE_ADD() -----Adds a specified time interval to a date:
---------------------------------------------------------
select date_add(STUDENT_ENROLLMENT_DATE,interval 10 day) day from student
select date_add(STUDENT_ENROLLMENT_DATE ,interval 1 month) month from student
select date_add(STUDENT_ENROLLMENT_DATE,interval 1 year) year from student


DATE_SUB() ------Subtracts a specified time interval from a date:
-----------------------------------------------------------------
select date_sub(STUDENT_ENROLLMENT_DATE,interval 10 day) day from student
select date_sub(STUDENT_ENROLLMENT_DATE,interval 1 month) month from student
select date_sub(STUDENT_ENROLLMENT_DATE,interval 1 year) year from student


DATEDIFF()-----Returns the number of days between two dates:
------------------------------------------------------------
select datediff (STUDENT_ENROLLMENT_DATE,'2011-02-21 00:00:00') as diff from student


timestamp:
==========
select current_timestamp();
select now();


TIMESTAMPADD() -----Adds a specified time interval to a timestamp:
------------------------------------------------------------------
select timestampadd(hour,5,STUDENT_ENROLLMENT_DATE ) as new_timestamp from student;
select timestampadd(minute,50,STUDENT_ENROLLMENT_DATE ) as new_timestamp from student;


TIMESTAMPDIFF()-----Returns the number of days between two dates:
-----------------------------------------------------------------
select timestampdiff(year,'1977-09-09',STUDENT_ENROLLMENT_DATE) as year_diff from student;
select timestampdiff(month,'1977-09-09',STUDENT_ENROLLMENT_DATE) as month_diff from student;
select timestampdiff(day,'2011-02-21 00:00:00',STUDENT_ENROLLMENT_DATE) as day_diff from student;

Joins:
------
create table student_course as (
select * from student s
left join
course c
on s.STUDENT_COURSE_ID = c.COURSE_ID
order by STUDENT_ID asc );

select * from student_course;


partition by:
------------
select *,row_number() over(partition by STUDENT_LOCATION) row_num from student_course;


Row Number: (order by is mandatory)
-----------
select *,row_number() over(partition by STUDENT_LOCATION order by STUDENT_COURSE_ID asc) row_num from student_course;


Rank :
------
select *,rank() over(partition by STUDENT_LOCATION order by STUDENT_COURSE_ID asc) rank_num from student_course;


Dense rank:
-----------
select *,dense_rank() over(partition by STUDENT_LOCATION order by STUDENT_COURSE_ID asc) dense_rank_num from student_course;
