6.What is the average order amount for customers who joined before ‘2023-03-25’?
================================================================================
select avg (total_price) average_order_amount from (
select o.order_id,sum(p.unit_price * o.quantity) total_price from orders o
join 
customer c
join
products p
on c.cust_id = o.cust_id
and o.prod_id = p.prod_id
where c.join_date < '2023-03-25'
group by o.order_id ) sub ;



7.Which product is the most popular among those customers who joined in April 2023?
==================================================================================
select p.prod_name,sum(o.quantity) total_prod_order from orders o
join
customer c
on c.cust_id = o.cust_id
join
products p
on o.prod_id = p.prod_id
where year(c.join_date) = 2023 and month(c.join_date) = 4
group by p.prod_name
order by total_prod_order desc;



8.What is the total revenue generated from orders placed in May 2023?
=====================================================================

where month(order_date) = 5 and year(order_date) = 2023;



9.What is the number of orders placed by customers who have spent more than $200 on the store?
==============================================================================================
select sum(num_ord_each_cust) total_num_ord  from (
select count( distinct order_id) num_ord_each_cust from orders o
join customer c on o.cust_id = c.cust_id
join products p on o.prod_id = p.prod_id
group by o.cust_id
having sum(o.quantity*p.unit_price) > 200 ) as sub;

14.Find the top 5 customers who have spent the most on their orders in the past 26 months?
==========================================================================================
select c.cust_id ,sum(o.quantity*p.unit_price) spent  from orders o
join
customer c on o.cust_id = c.cust_id
join
products p on p.prod_id = o.prod_id
where o.order_date >= date_sub(curdate(), interval 26 month)
group by c.cust_id
order by spent desc
limit 5;



15.What is the average order amount for customers who have purchased a “Laptop”?
================================================================================
select sum.cust_id,avg(total_amt) from 
(select distinct o.cust_id,o.order_id,sum(o.quantity*p.unit_price) total_amt from orders o
join customer c on c.cust_id = o.cust_id
join products p on p.prod_id = o.prod_id
group by o.cust_id,o.order_id ) sum

join (select o.order_id,o.cust_id from orders o
join customer c on c.cust_id = o.cust_id
join products p on p.prod_id = o.prod_id
where p.prod_name = 'Laptop') lap
on sum.order_id= lap.order_id
group by sum.cust_id;

26.Query to format order_date as day,month ?
============================================

- %b → Abbreviated month name (Jan, Feb, Mar…)
- %y → Two‑digit year
